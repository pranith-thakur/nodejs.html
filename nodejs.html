// Import the built-in http module const http = require('http'); // Define the server port and hostname const hostname = '127.0.0.1'; const port = 3000; // Create an HTTP server that responds with "Hello, World!" const server = http.createServer((req, res) => { res.statusCode = 200; // HTTP Status 200 (OK) res.setHeader('Content-Type', 'text/plain'); res.end('Hello, World!\n'); // Response body }); // Start the server and listen on the specified port server.listen(port, hostname, () => { console.log(`Server running at http://${hostname}:${port}/`); });
